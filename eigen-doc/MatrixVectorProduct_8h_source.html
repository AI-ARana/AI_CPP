378 template<typename Index, typename LhsScalar, typename LhsMapper, typename RhsScalar, typename RhsMapper, typename ResScalar>
379 EIGEN_STRONG_INLINE void gemv_col(
380  Index rows, Index cols,
381  const LhsMapper& alhs,
382  const RhsMapper& rhs,
383  ResScalar* res, Index resIncr,
384  ResScalar alpha)
385 {
386  typedef gemv_traits<LhsScalar, RhsScalar> Traits;
387 
388  typedef typename Traits::LhsPacket LhsPacket;
389  typedef typename Traits::RhsPacket RhsPacket;
390  typedef typename Traits::ResPacket ResPacket;
391 
392  EIGEN_UNUSED_VARIABLE(resIncr);
393  eigen_internal_assert(resIncr == 1);
394 
395  // The following copy tells the compiler that lhs's attributes are not modified outside this function
396  // This helps GCC to generate proper code.
397  LhsMapper lhs(alhs);
398  RhsMapper rhs2(rhs);
399 
400  conj_helper<LhsScalar, RhsScalar, false, false> cj;
401  conj_helper<LhsPacket, RhsPacket, false, false> pcj;
402 
403  const Index lhsStride = lhs.stride();
404  // TODO: for padded aligned inputs, we could enable aligned reads
405  enum {
406  LhsAlignment = Unaligned,
407  ResPacketSize = Traits::ResPacketSize,
408  LhsPacketSize = Traits::LhsPacketSize,
409  RhsPacketSize = Traits::RhsPacketSize,
410  };
411 
412 #ifndef GCC_ONE_VECTORPAIR_BUG
413  const Index n8 = rows - 8 * ResPacketSize + 1;
414  const Index n4 = rows - 4 * ResPacketSize + 1;
415  const Index n2 = rows - 2 * ResPacketSize + 1;
416 #endif
417  const Index n1 = rows - 1 * ResPacketSize + 1;
418 #ifdef EIGEN_POWER_USE_GEMV_PREFETCH
419  const Index prefetch_dist = 64 * LhsPacketSize;
420 #endif
421 
422  // TODO: improve the following heuristic:
423  const Index block_cols = cols < 128 ? cols : (lhsStride * sizeof(LhsScalar) < 16000 ? 16 : 8);
424  ResPacket palpha = pset1<ResPacket>(alpha);
425 
426  for (Index j2 = 0; j2 < cols; j2 += block_cols)
427  {
428  Index jend = numext::mini(j2 + block_cols, cols);
429  Index i = 0;
430  ResPacket c0, c1, c2, c3, c4, c5, c6, c7;
431 #ifdef USE_GEMV_MMA
432  __vector_quad e0, e1, e2, e3, e4, e5, e6, e7;
433  PacketBlock<ResPacket, 4> result0, result1, result2, result3, result4, result5, result6, result7;
434  GEMV_UNUSED(8, e)
435  GEMV_UNUSED(8, result)
436  GEMV_UNUSED_EXTRA(1, c)
437 #endif
438 #ifndef GCC_ONE_VECTORPAIR_BUG
439  while (i < n8)
440  {
441  GEMV_PROCESS_COL(8)
442  }
443  if (i < n4)
444  {
445  GEMV_PROCESS_COL(4)
446  }
447  if (i < n2)
448  {
449  GEMV_PROCESS_COL(2)
450  }
451  if (i < n1)
452 #else
453  while (i < n1)
454 #endif
455  {
456  GEMV_PROCESS_COL_ONE(1)
457  }
458  for (;i < rows;++i)
459  {
460  ResScalar d0(0);
461  Index j = j2;
462  do {
463  d0 += cj.pmul(lhs(i, j), rhs2(j, 0));
464  } while (++j < jend);
465  res[i] += alpha * d0;
466  }
467  }
468 }
