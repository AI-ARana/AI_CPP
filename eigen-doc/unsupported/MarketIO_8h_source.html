122 inline bool getMarketHeader(const std::string& filename, int& sym, bool& iscomplex, bool& isdense)
123 {
124  sym = 0;
125  iscomplex = false;
126  isdense = false;
127  std::ifstream in(filename.c_str(),std::ios::in);
128  if(!in)
129  return false;
130 
131  std::string line;
132  // The matrix header is always the first line in the file
133  std::getline(in, line); eigen_assert(in.good());
134 
135  std::stringstream fmtline(line);
136  std::string substr[5];
137  fmtline>> substr[0] >> substr[1] >> substr[2] >> substr[3] >> substr[4];
138  if(substr[2].compare("array") == 0) isdense = true;
139  if(substr[3].compare("complex") == 0) iscomplex = true;
140  if(substr[4].compare("symmetric") == 0) sym = Symmetric;
141  else if (substr[4].compare("Hermitian") == 0) sym = SelfAdjoint;
142 
143  return true;
144 }
154 template<typename SparseMatrixType>
155 bool loadMarket(SparseMatrixType& mat, const std::string& filename)
156 {
157  typedef typename SparseMatrixType::Scalar Scalar;
158  typedef typename SparseMatrixType::StorageIndex StorageIndex;
159  std::ifstream input(filename.c_str(),std::ios::in);
160  if(!input)
161  return false;
162 
163  char rdbuffer[4096];
164  input.rdbuf()->pubsetbuf(rdbuffer, 4096);
165 
166  const int maxBuffersize = 2048;
167  char buffer[maxBuffersize];
168 
169  bool readsizes = false;
170 
171  typedef Triplet<Scalar,StorageIndex> T;
172  std::vector<T> elements;
173 
174  Index M(-1), N(-1), NNZ(-1);
175  Index count = 0;
176  while(input.getline(buffer, maxBuffersize))
177  {
178  // skip comments
179  //NOTE An appropriate test should be done on the header to get the symmetry
180  if(buffer[0]=='%')
181  continue;
182 
183  if(!readsizes)
184  {
185  std::stringstream line(buffer);
186  line >> M >> N >> NNZ;
187  if(M > 0 && N > 0)
188  {
189  readsizes = true;
190  mat.resize(M,N);
191  mat.reserve(NNZ);
192  }
193  }
194  else
195  {
196  StorageIndex i(-1), j(-1);
197  Scalar value;
198  internal::GetMarketLine(buffer, i, j, value);
199 
200  i--;
201  j--;
202  if(i>=0 && j>=0 && i<M && j<N)
203  {
204  ++count;
205  elements.push_back(T(i,j,value));
206  }
207  else
208  {
209  std::cerr << "Invalid read: " << i << "," << j << "\n";
210  return false;
211  }
212  }
213  }
214 
215  mat.setFromTriplets(elements.begin(), elements.end());
216  if(count!=NNZ){
217  std::cerr << count << "!=" << NNZ << "\n";
218  return false;
219  }
220  input.close();
221  return true;
222 }

234 template<typename DenseType>
235 bool loadMarketDense(DenseType& mat, const std::string& filename)
236 {
237  typedef typename DenseType::Scalar Scalar;
238  std::ifstream in(filename.c_str(), std::ios::in);
239  if(!in)
240  return false;
241 
242  std::string line;
243  Index rows(0), cols(0);
244  do
245  { // Skip comments
246  std::getline(in, line); eigen_assert(in.good());
247  } while (line[0] == '%');
248  std::istringstream newline(line);
249  newline >> rows >> cols;
250 
251  bool sizes_not_positive=(rows<1 || cols<1);
252  bool wrong_input_rows = (DenseType::MaxRowsAtCompileTime != Dynamic && rows > DenseType::MaxRowsAtCompileTime) ||
253  (DenseType::RowsAtCompileTime!=Dynamic && rows!=DenseType::RowsAtCompileTime);
254  bool wrong_input_cols = (DenseType::MaxColsAtCompileTime != Dynamic && cols > DenseType::MaxColsAtCompileTime) ||
255  (DenseType::ColsAtCompileTime!=Dynamic && cols!=DenseType::ColsAtCompileTime);
256 
257  if(sizes_not_positive || wrong_input_rows || wrong_input_cols){
258  if(sizes_not_positive){
259  std::cerr<< "non-positive row or column size in file" << filename << "\n";
260  }else{
261  std::cerr<< "Input matrix can not be resized to"<<rows<<" x "<<cols<< "as given in " << filename << "\n";
262  }
263  in.close();
264  return false;
265  }
266 
267  mat.resize(rows,cols);
268  Index row = 0;
269  Index col = 0;
270  Index n=0;
271  Scalar value;
272  while ( std::getline(in, line) && (row < rows) && (col < cols)){
273  internal::GetDenseElt(line, value);
274  //matrixmarket format is column major
275  mat(row,col) = value;
276  row++;
277  if(row==rows){
278  row=0;
279  col++;
280  }
281  n++;
282  }
283  in.close();
284  if (n!=mat.size()){
285  std::cerr<< "Unable to read all elements from file " << filename << "\n";
286  return false;
287  }
288  return true;
289 }

310 template<typename SparseMatrixType>
311 bool saveMarket(const SparseMatrixType& mat, const std::string& filename, int sym = 0)
312 {
313  typedef typename SparseMatrixType::Scalar Scalar;
314  typedef typename SparseMatrixType::RealScalar RealScalar;
315  std::ofstream out(filename.c_str(),std::ios::out);
316  if(!out)
317  return false;
318 
319  out.flags(std::ios_base::scientific);
320  out.precision(std::numeric_limits<RealScalar>::digits10 + 2);
321  std::string header;
322  internal::putMarketHeader<Scalar>(header, sym);
323  out << header << std::endl;
324  out << mat.rows() << " " << mat.cols() << " " << mat.nonZeros() << "\n";
325  int count = 0;
326  for(int j=0; j<mat.outerSize(); ++j)
327  for(typename SparseMatrixType::InnerIterator it(mat,j); it; ++it)
328  {
329  ++ count;
330  internal::PutMatrixElt(it.value(), it.row()+1, it.col()+1, out);
331  }
332  out.close();
333  return true;
334 }

347 template<typename DenseType>
348 bool saveMarketDense (const DenseType& mat, const std::string& filename)
349 {
350  typedef typename DenseType::Scalar Scalar;
351  typedef typename DenseType::RealScalar RealScalar;
352  std::ofstream out(filename.c_str(),std::ios::out);
353  if(!out)
354  return false;
355 
356  out.flags(std::ios_base::scientific);
357  out.precision(std::numeric_limits<RealScalar>::digits10 + 2);
358  if(internal::is_same<Scalar, std::complex<float> >::value || internal::is_same<Scalar, std::complex<double> >::value)
359  out << "%%MatrixMarket matrix array complex general\n";
360  else
361  out << "%%MatrixMarket matrix array real general\n";
362  out << mat.rows() << " "<< mat.cols() << "\n";
363  for (Index i=0; i < mat.cols(); i++){
364  for (Index j=0; j < mat.rows(); j++){
365  internal::putDenseElt(mat(j,i), out);
366  }
367  }
368  out.close();
369  return true;
370 }
